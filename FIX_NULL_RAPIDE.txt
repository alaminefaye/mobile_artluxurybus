━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PROBLÈME :
  L'API Laravel renvoyait parfois null pour :
  • file_size
  • display_order
  • views_count
  • duration_formatted

SOLUTION :
  Ajout de valeurs par défaut si null :
  • file_size → 0
  • display_order → 0
  • views_count → 0
  • duration_formatted → 'N/A'
  • file_size_formatted → '0 B'

CODE :
  AVANT ❌ :
    fileSize: json['file_size'] as int,  // Crash si null

  APRÈS ✅ :
    fileSize: (json['file_size'] as num?)?.toInt() ?? 0,
